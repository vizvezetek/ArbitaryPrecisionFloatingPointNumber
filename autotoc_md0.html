Fpn e("0.111111");
double i = -22.9911;
double j = 3333.111189;
double k = -123.234;
double l = 0.456;
double m = 0.111111;
Fpn f = a*b*c*d*e;

cout << "FPN: " << f.toString() << endl;
cout << "double: " << std::setprecision(50) << o << endl;
return 0;
}
